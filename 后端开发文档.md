#### 3.4.2 distribution.py

```python
import numpy as np
from app.models.kl import CurvePoint

def generate_normal_curve(
    mean: float,
    std: float,
    x_range: tuple[float, float] = (-4.0, 4.0),
    num_points: int = 200,
) -> list[CurvePoint]:
    """
    生成正态分布曲线采样点。
    来源：算法B3
    """
    x_values = np.linspace(x_range[0], x_range[1], num_points)
    y_values = (1 / (std * np.sqrt(2 * np.pi))) * np.exp(-0.5 * ((x_values - mean) / std) ** 2)
    return [CurvePoint(x=round4(x), y=round4(y)) for x, y in zip(x_values, y_values)]

def calculate_overlap_area(
    mean1: float, std1: float,
    mean2: float, std2: float,
    x_range: tuple[float, float] = (-4.0, 4.0),
    num_points: int = 1000,
) -> float:
    """
    计算两个正态分布的重叠面积（数值积分）。
    重叠面积 = ∫ min(f1(x), f2(x)) dx
    """
    x_values = np.linspace(x_range[0], x_range[1], num_points)
    y1 = (1 / (std1 * np.sqrt(2 * np.pi))) * np.exp(-0.5 * ((x_values - mean1) / std1) ** 2)
    y2 = (1 / (std2 * np.sqrt(2 * np.pi))) * np.exp(-0.5 * ((x_values - mean2) / std2) ** 2)
    overlap = np.minimum(y1, y2)
    dx = (x_range[1] - x_range[0]) / (num_points - 1)
    return round4(float(np.trapz(overlap, dx=dx)))
```

### 4.1 GAE计算 — `app/algorithm/gae.py`

**来源：算法C1（完整GAE计算）、算法C2（单步GAE计算）**

#### 4.1.1 compute_gae — 完整GAE计算

```python
import numpy as np
from app.utils.numeric import round4, check_overflow

def compute_gae(
    rewards: list[float],
    values: list[float],
    gamma: float = 0.99,
    lam: float = 0.95,
) -> tuple[list[float], list[float]]:
    """
    GAE从后向前动态规划计算。
    来源：算法C1，知乎文章8.3节

    公式：
        δ_t = r_t + γ·V_{t+1} - V_t
        A_t = δ_t + γ·λ·A_{t+1}

    Args:
        rewards: 即时奖励数组 [r_0, r_1, ..., r_T]
        values: Critic预测价值数组 [V_0, V_1, ..., V_T]
        gamma: 折扣因子（默认0.99）
        lam: GAE权衡因子λ（默认0.95）

    Returns:
        (advantages, returns): 优势函数数组和Returns数组，均按原始时间顺序排列

    Raises:
        ValueError: rewards和values长度不一致时
    """
    if len(rewards) != len(values):
        raise ValueError("rewards和values长度必须相同")
    if len(rewards) < 2:
        raise ValueError("序列长度必须≥2")

    length = len(rewards)
    advantages = [0.0] * length
    last_gae_lam = 0.0

    for t in range(length - 1, -1, -1):
        next_value = values[t + 1] if t < length - 1 else 0.0
        delta = rewards[t] + gamma * next_value - values[t]
        last_gae_lam = delta + gamma * lam * last_gae_lam
        advantages[t] = last_gae_lam

    returns = [round4(adv + val) for adv, val in zip(advantages, values)]
    advantages = [round4(a) for a in advantages]

    if any(check_overflow(a) for a in advantages):
        raise ArithmeticError("GAE计算结果溢出")

    return advantages, returns
```

#### 4.1.2 compute_gae_step — 单步GAE计算

```python
def compute_gae_step(
    t: int,
    rewards: list[float],
    values: list[float],
    gamma: float,
    lam: float,
    next_advantage: float = 0.0,
) -> tuple[float, float, dict]:
    """
    单步GAE计算，返回计算详情用于前端打字机效果展示。
    来源：算法C2

    Args:
        t: 当前回溯时刻
        rewards: 即时奖励数组
        values: Critic预测价值数组
        gamma: 折扣因子
        lam: GAE权衡因子
        next_advantage: 下一时刻已计算的优势值（默认0.0）

    Returns:
        (delta, advantage, calculation):
            delta: TD error
            advantage: 当前时刻优势值
            calculation: 计算详情字典

    Raises:
        ValueError: t超出有效范围时
    """
    if t < 0 or t >= len(rewards):
        raise ValueError(f"t={t}超出有效范围[0, {len(rewards)-1}]")

    next_value = values[t + 1] if t < len(values) - 1 else 0.0

    delta = rewards[t] + gamma * next_value - values[t]
    advantage = delta + gamma * lam * next_advantage

    calculation = {
        "td_error": {
            "formula": "δ_t = r_t + γ·V_{t+1} - V_t",
            "substitution": f"δ_{t} = {rewards[t]:.4f} + {gamma}×{next_value:.4f} - {values[t]:.4f}",
            "result": round4(delta),
        },
        "gae": {
            "formula": "A_t = δ_t + γ·λ·A_{t+1}",
            "substitution": f"A_{t} = {delta:.4f} + {gamma}×{lam}×{next_advantage:.4f}",
            "result": round4(advantage),
        },
    }

    return round4(delta), round4(advantage), calculation
```

#### 4.1.3 get_variance_bias_interpretation — 方差-偏差解读

```python
def get_variance_bias_interpretation(lam: float) -> dict:
    """
    根据λ值返回方差-偏差平衡解读。
    来源：算法C1，知乎文章8.3节

    λ接近0：高偏差低方差（更信任Critic估计）
    λ接近1：低偏差高方差（更信任实际采样）
    """
    if lam < 0.3:
        return {
            "bias": "高偏差",
            "variance": "低方差",
            "description": "更信任Critic的估计，采样负担小，但可能引入系统性偏差",
        }
    elif lam <= 0.7:
        return {
            "bias": "中等偏差",
            "variance": "中等方差",
            "description": "在偏差和方差之间取得平衡",
        }
    else:
        return {
            "bias": "低偏差",
            "variance": "高方差",
            "description": "更信任实际采样结果，偏差小但需要更多采样",
        }
```

### 4.2 PPO-Clip Actor Loss — `app/algorithm/ppo_clip.py`

**来源：算法D1，知乎文章8.5节**

```python
import numpy as np
from app.utils.numeric import round4, check_overflow

def calculate_actor_loss(
    log_probs: list[float],
    old_log_probs: list[float],
    advantages: list[float],
    clip_range: float = 0.2,
) -> tuple[float, list[float], list[float]]:
    """
    PPO-Clip Actor Loss计算。
    来源：算法D1，知乎文章8.5节

    公式：
        ratio = exp(log_probs - old_log_probs)
        unclipped = ratio · A
        clipped = clip(ratio, 1-ε, 1+ε) · A
        loss = -mean(min(unclipped, clipped))

    核心思想：
        当A > 0时，ratio有上限1+ε，不能一味提升π_θ(a_t|s_t)
        当A < 0时，ratio有下限1-ε，不能一味降低π_θ(a_t|s_t)

    Args:
        log_probs: 当前Actor的log概率
        old_log_probs: 旧Actor的log概率（生成经验时）
        advantages: GAE计算的优势值
        clip_range: Clip范围ε（默认0.2）

    Returns:
        (actor_loss, ratios, clipped_ratios):
            actor_loss: Actor Loss值
            ratios: 重要性采样比值数组
            clipped_ratios: 裁剪后的比值数组

    Raises:
        ValueError: 输入数组长度不一致时
    """
    if not (len(log_probs) == len(old_log_probs) == len(advantages)):
        raise ValueError("log_probs、old_log_probs、advantages长度必须相同")

    ratios = [round4(np.exp(lp - olp)) for lp, olp in zip(log_probs, old_log_probs)]

    clipped_ratios = [
        round4(max(1 - clip_range, min(1 + clip_range, r))) for r in ratios
    ]

    unclipped_objectives = [r * a for r, a in zip(ratios, advantages)]
    clipped_objectives = [cr * a for cr, a in zip(clipped_ratios, advantages)]

    losses = [-min(u, c) for u, c in zip(unclipped_objectives, clipped_objectives)]
    actor_loss = round4(sum(losses) / len(losses))

    if check_overflow(actor_loss):
        raise ArithmeticError("Actor Loss计算结果溢出")

    return actor_loss, ratios, clipped_ratios
```

### 4.3 Critic Loss — `app/algorithm/critic_loss.py`

**来源：算法D2，知乎文章8.7节**

```python
import numpy as np
from app.utils.numeric import round4, check_overflow

def calculate_critic_loss(
    values: list[float],
    old_values: list[float],
    returns: list[float],
    clip_range_value: float = 0.2,
) -> float:
    """
    Critic Loss计算（Value Clipping）。
    来源：算法D2，知乎文章8.7节

    公式：
        V_t^{CLIP} = clip(V_t^{new}, V_t^{old} - ε, V_t^{old} + ε)
        L(V_φ) = E_t[max[(V_t^{new} - R_t)², (V_t^{CLIP} - R_t)²]] × 0.5

    Args:
        values: 当前Critic预测的价值 V_t^{new}
        old_values: 旧Critic预测的价值 V_t^{old}
        returns: GAE计算的returns（R_t = A_t^{GAE} + V_t^{old}）
        clip_range_value: Value裁剪范围（默认0.2）

    Returns:
        critic_loss: Critic Loss值

    Raises:
        ValueError: 输入数组长度不一致时
    """
    if not (len(values) == len(old_values) == len(returns)):
        raise ValueError("values、old_values、returns长度必须相同")

    values_clipped = [
        max(old_v - clip_range_value, min(old_v + clip_range_value, v))
        for v, old_v in zip(values, old_values)
    ]

    vf_loss1 = [(v - r) ** 2 for v, r in zip(values, returns)]
    vf_loss2 = [(vc - r) ** 2 for vc, r in zip(values_clipped, returns)]

    vf_loss = [max(l1, l2) for l1, l2 in zip(vf_loss1, vf_loss2)]
    critic_loss = round4(sum(vf_loss) / len(vf_loss) * 0.5)

    if check_overflow(critic_loss):
        raise ArithmeticError("Critic Loss计算结果溢出")

    return critic_loss
```

**来源：算法B2（KL高斯计算）、算法B1（PPO-Penalty解读）**

#### 4.4.1 calculate_kl_gaussian — 两个正态分布间KL散度

```python
import numpy as np
from app.utils.numeric import round4, check_overflow

def calculate_kl_gaussian(
    mu1: float, sigma1: float,
    mu2: float, sigma2: float,
) -> float:
    """
    两个正态分布之间的KL散度解析公式。
    来源：算法B2

    公式：
        KL(N1‖N2) = log(σ2/σ1) + (σ1² + (μ1-μ2)²)/(2·σ2²) - 0.5

    其中 N1 = Reference, N2 = Actor

    Args:
        mu1, sigma1: 分布1参数（Reference）
        mu2, sigma2: 分布2参数（Actor）

    Returns:
        kl_divergence: KL散度值

    Raises:
        ValueError: sigma1或sigma2 ≤ 0
    """
    if sigma1 <= 0 or sigma2 <= 0:
        raise ValueError("标准差必须大于0")

    ratio = sigma2 / sigma1
    diff = mu1 - mu2
    kl = np.log(ratio) + (sigma1 ** 2 + diff ** 2) / (2 * sigma2 ** 2) - 0.5
    result = round4(float(kl))

    if check_overflow(result):
        raise ArithmeticError("KL散度计算结果溢出")

    return result
```

#### 4.4.2 calculate_kl_per_token — 单Token KL散度

```python
def calculate_kl_per_token(
    log_probs: list[float],
    ref_log_probs: list[float],
) -> list[float]:
    """
    单Token KL散度近似计算。
    来源：算法A2，知乎文章8.6节 PPO-Penalty

    公式：kl_div = log_prob - ref_log_prob

    Args:
        log_probs: Actor的log概率数组
        ref_log_probs: Reference的log概率数组

    Returns:
        kl_divs: 每个Token的KL散度值数组
    """
    if len(log_probs) != len(ref_log_probs):
        raise ValueError("log_probs和ref_log_probs长度必须相同")

    return [round4(lp - rlp) for lp, rlp in zip(log_probs, ref_log_probs)]
```

### 4.6 Reward计算 — `app/algorithm/reward.py`

**来源：算法A3，报告4.3节（DeepSpeed-Chat实现）**

```python
from app.utils.numeric import round4

def calculate_rewards(
    kl_divs: list[float],
    kl_ctl: float,
    r_global: float,
    is_last_token: list[bool],
) -> list[float]:
    """
    R_t设计（DeepSpeed-Chat实现）。
    来源：算法A3，报告4.3节

    公式：
        当 t ≠ T: R_t = -kl_ctl × kl_div_t
        当 t = T: R_t = -kl_ctl × kl_div_t + R_global

    Args:
        kl_divs: KL散度数组
        kl_ctl: KL惩罚系数β
        r_global: 全局奖励（由Reward Model给出）
        is_last_token: 是否为序列最后一个Token的布尔数组

    Returns:
        rewards: 每个Token的即时奖励数组
    """
    rewards = []
    for kl, is_last in zip(kl_divs, is_last_token):
        kl_penalty = -kl_ctl * kl
        if is_last:
            rewards.append(round4(kl_penalty + r_global))
        else:
            rewards.append(round4(kl_penalty))
    return rewards
```

### 4.7 Token生成流程 — `app/algorithm/token_generation.py`

**来源：算法A1**

```python
import numpy as np
from app.utils.numeric import round4

VOCAB = ["I", "am", "an", "AI", "model", "assistant", "the", "hello", "world", "deep", "learning", "language"]

def generate_token_log_probs(
    context_tokens: list[str],
    actor_params: dict,
    vocab: list[str] | None = None,
) -> tuple[str, float]:
    """
    基于当前上下文生成下一个Token及其log概率。
    来源：算法A1

    简化模拟：使用softmax over正态分布采样的logits

    Args:
        context_tokens: 当前已生成的Token序列
        actor_params: Actor模型参数 {"mean": float, "std": float}
        vocab: 词表（默认使用内置词表）

    Returns:
        (token, log_prob): 生成的Token文本和log概率
    """
    if vocab is None:
        vocab = VOCAB

    rng = np.random.default_rng(hash(tuple(context_tokens)) % (2**31))
    logits = rng.normal(actor_params["mean"], actor_params["std"], size=len(vocab))
    probs = np.exp(logits - np.max(logits))
    probs = probs / probs.sum()

    token_idx = rng.choice(len(vocab), p=probs)
    token = vocab[token_idx]
    log_prob = round4(float(np.log(probs[token_idx])))

    return token, log_prob


def generate_ref_log_prob(
    token: str,
    ref_params: dict,
    vocab: list[str] | None = None,
) -> float:
    """
    计算Reference模型对给定Token的log概率。
    来源：算法A1

    Args:
        token: Token文本
        ref_params: Reference模型参数 {"mean": float, "std": float}
        vocab: 词表

    Returns:
        ref_log_prob: Reference的log概率
    """
    if vocab is None:
        vocab = VOCAB

    rng = np.random.default_rng(hash(token) % (2**31))
    logits = rng.normal(ref_params["mean"], ref_params["std"], size=len(vocab))
    probs = np.exp(logits - np.max(logits))
    probs = probs / probs.sum()

    token_idx = vocab.index(token) if token in vocab else 0
    return round4(float(np.log(probs[token_idx])))
```

### 4.8 完整PPO训练步骤 — `app/algorithm/training.py`

**来源：算法D4（完整PPO训练步骤）、算法D5（模型参数更新）**

```python
import numpy as np
from app.utils.numeric import round4, check_overflow
from app.algorithm.token_generation import generate_token_log_probs, generate_ref_log_prob
from app.algorithm.kl_divergence import calculate_kl_per_token
from app.algorithm.reward import calculate_rewards
from app.algorithm.gae import compute_gae
from app.algorithm.ppo_clip import calculate_actor_loss
from app.algorithm.critic_loss import calculate_critic_loss
from app.algorithm.returns import calculate_returns

def ppo_training_step(
    actor_params: dict,
    critic_params: dict,
    ref_params: dict,
    hyperparams: dict,
    batch_data: dict | None = None,
) -> dict:
    """
    完整PPO训练步骤。
    来源：算法D4，知乎文章第8节

    流程：
        1. 生成经验（使用old策略）
        2. 保存old值（重要性采样）
        3. 计算KL散度
        4. GAE计算优势和returns
        5. PPO多轮更新
        6. 计算平均Reward

    Args:
        actor_params: Actor模型参数 {"mean": float, "std": float}
        critic_params: Critic模型参数 {"mean": float, "std": float, "weights": list}
        ref_params: Reference模型参数（固定不变）
        hyperparams: 训练超参数字典
        batch_data: 训练批次数据（可选，用于复现）

    Returns:
        训练指标字典
    """
    kl_ctl = hyperparams.get("kl_ctl", 0.1)
    gamma = hyperparams.get("gamma", 0.99)
    lam = hyperparams.get("lam", 0.95)
    clip_range = hyperparams.get("clip_range", 0.2)
    learning_rate = hyperparams.get("learning_rate", 0.03)
    ppo_epochs = hyperparams.get("ppo_epochs", 3)
    max_tokens = hyperparams.get("max_tokens", 6)

    # 1. 生成经验
    tokens = []
    log_probs = []
    ref_log_probs_list = []
    values = []
    r_global = round4(np.random.uniform(0.5, 1.0))

    for i in range(max_tokens):
        token, log_prob = generate_token_log_probs(tokens, actor_params)
        ref_log_prob = generate_ref_log_prob(token, ref_params)

        # Critic预测价值（简化：基于参数的线性变换 + 噪声）
        v_t = round4(
            critic_params.get("mean", 0.5)
            + np.random.normal(0, 0.1)
        )

        tokens.append(token)
        log_probs.append(log_prob)
        ref_log_probs_list.append(ref_log_prob)
        values.append(v_t)

    # 2. 保存old值
    old_log_probs = log_probs[:]
    old_values = values[:]

    # 3. 计算KL散度
    kl_divs = calculate_kl_per_token(log_probs, ref_log_probs_list)
    avg_kl = round4(sum(kl_divs) / len(kl_divs))

    # 4. 计算Reward
    is_last_token = [False] * (max_tokens - 1) + [True]
    rewards = calculate_rewards(kl_divs, kl_ctl, r_global, is_last_token)

    # 5. GAE计算
    advantages, returns = compute_gae(rewards, values, gamma, lam)

    # 6. PPO多轮更新
    current_log_probs = log_probs[:]
    current_values = values[:]

    for epoch in range(ppo_epochs):
        actor_loss, ratios, clipped_ratios = calculate_actor_loss(
            current_log_probs, old_log_probs, advantages, clip_range
        )
        critic_loss = calculate_critic_loss(
            current_values, old_values, returns, clip_range
        )

        # 参数更新（简化模拟）
        actor_params["mean"] -= actor_loss * learning_rate * 0.01
        actor_params["std"] = max(0.1, actor_params["std"] - abs(actor_loss) * learning_rate * 0.001)

        # 模拟参数更新后的新log_probs和values
        current_log_probs = [
            round4(lp + np.random.normal(0, learning_rate * 0.5))
            for lp in current_log_probs
        ]
        current_values = [
            round4(v + np.random.normal(0, learning_rate * 0.3))
            for v in current_values
        ]

    # 7. 计算平均Reward
    avg_reward = round4(sum(rewards) / len(rewards))

    # 溢出检测
    if any(check_overflow(v) for v in [actor_loss, critic_loss, avg_kl, avg_reward]):
        raise ArithmeticError("训练计算结果溢出")

    return {
        "actor_loss": actor_loss,
        "critic_loss": critic_loss,
        "kl_div": avg_kl,
        "reward": avg_reward,
        "advantages": advantages,
        "returns": returns,
    }
```

### 5.5 WebSocket训练接口 — `app/api/ws_training.py`

```python
import asyncio
import json
from fastapi import WebSocket, WebSocketDisconnect
from app.models.training import TrainingHyperparams, TrainingMetrics, AchievementUnlock, AchievementType
from app.algorithm.training import ppo_training_step
from app.utils.numeric import check_overflow

# 全局训练状态（同一时间只允许一个训练会话）
class TrainingSession:
    def __init__(self):
        self.state: str = "idle"  # idle | running | paused | completed | error
        self.current_epoch: int = 0
        self.max_epochs: int = 100
        self.hyperparams: dict = TrainingHyperparams().model_dump()

async def handle_training_websocket(websocket: WebSocket):
    """
    WebSocket训练接口处理。
    来源：API接口规范第四章

    消息格式：
        客户端→服务端: {action: string, params?: object}
        服务端→客户端: {type: string, data: object}
    """
    await websocket.accept()

    # 发送初始状态
    await websocket.send_json({
        "type": "status",
        "data": {
            "state": session.state,
            "epoch": session.current_epoch,
            "max_epochs": session.max_epochs,
        },
    })

    try:
        while True:
            data = await websocket.receive_json()
            action = data.get("action")

            if action == "start":
                await handle_start(websocket, data.get("params"))
            elif action == "pause":
                await handle_pause(websocket)
            elif action == "resume":
                await handle_resume(websocket)
            elif action == "step":
                await handle_step(websocket)
            elif action == "reset":
                await handle_reset(websocket)
            elif action == "update_params":
                await handle_update_params(websocket, data.get("params"))

    except WebSocketDisconnect:
        session.state = "idle"

async def run_training_loop(websocket: WebSocket):

    while session.state == "running" and session.current_epoch < session.max_epochs:
        await execute_one_epoch(websocket)
        await asyncio.sleep(0.1)  # 最小间隔100ms，防止前端渲染压力

    if session.current_epoch >= session.max_epochs:
        session.state = "completed"
        await websocket.send_json({
            "type": "status",
            "data": {
                "state": "completed",
                "epoch": session.current_epoch,
                "max_epochs": session.max_epochs,
            },
        })

async def execute_one_epoch(websocket: WebSocket):
    """执行一个Epoch的训练"""
    try:
        session.current_epoch += 1
        metrics = ppo_training_step(
            actor_params=session.actor_params,
            critic_params=session.critic_params,
            ref_params=session.ref_params,
            hyperparams=session.hyperparams,
        )

        epoch_data = {
            "epoch": session.current_epoch,
            "actor_loss": metrics["actor_loss"],
            "critic_loss": metrics["critic_loss"],
            "kl_div": metrics["kl_div"],
            "reward": metrics["reward"],
            "advantages": metrics["advantages"],
            "returns": metrics["returns"],
        }
        session.history.append(epoch_data)

        await websocket.send_json({"type": "epoch", "data": epoch_data})

        # 成就检测
        achievements = check_achievements(session.history, epoch_data)
        for ach in achievements:
            await websocket.send_json({"type": "achievement", "data": ach})

    except ArithmeticError as e:
        session.state = "error"
        await websocket.send_json({
            "type": "error",
            "data": {"code": "TRAINING_DIVERGED", "message": str(e)},
        })

def check_achievements(history: list[dict], current: dict) -> list[dict]:
    """
    成就判定逻辑。
    来源：算法D5
    """
    achievements = []

    # 1. 首次收敛
    if "first_converge" not in session.unlocked_achievements and current["actor_loss"] < 0.5:
        session.unlocked_achievements.add("first_converge")
        achievements.append({
            "achievement": "first_converge",
            "description": "Actor Loss首次降到0.5以下",
            "unlocked_at_epoch": current["epoch"],
        })

    # 2. KL稳定
    if "kl_stable" not in session.unlocked_achievements and len(history) >= 10:
        recent_kl = [h["kl_div"] for h in history[-10:]]
        if all(0.1 <= kl <= 0.3 for kl in recent_kl):
            session.unlocked_achievements.add("kl_stable")
            achievements.append({
                "achievement": "kl_stable",
                "description": "KL散度连续10个Epoch在[0.1, 0.3]之间",
                "unlocked_at_epoch": current["epoch"],
            })

    # 3. 连续优化
    if "continuous_improve" not in session.unlocked_achievements and len(history) >= 20:
        recent_losses = [h["actor_loss"] for h in history[-20:]]
        if all(recent_losses[i] <= recent_losses[i - 1] for i in range(1, len(recent_losses))):
            session.unlocked_achievements.add("continuous_improve")
            achievements.append({
                "achievement": "continuous_improve",
                "description": "Actor Loss连续20个Epoch下降",
                "unlocked_at_epoch": current["epoch"],
            })

    # 4. 精准调参
    if "precise_tuning" not in session.unlocked_achievements:
        if current["actor_loss"] < 0.3 and current["kl_div"] < 0.2:
            session.unlocked_achievements.add("precise_tuning")
            achievements.append({
                "achievement": "precise_tuning",
                "description": "Actor Loss < 0.3 且 KL散度 < 0.2",
                "unlocked_at_epoch": current["epoch"],
            })

    # 5. 驯化大师
    if "master" not in session.unlocked_achievements:
        if current["actor_loss"] < 0.1 and current["kl_div"] < 0.2 and current["reward"] > 0.8:
            session.unlocked_achievements.add("master")
            achievements.append({
                "achievement": "master",
                "description": "Actor Loss < 0.1 且 KL散度 < 0.2 且 Reward > 0.8",
                "unlocked_at_epoch": current["epoch"],
            })

    return achievements
```

```python
        assert abs(advantages[5] - (-0.1000)) < 1e-3, f"A_5期望-0.1000，实际{advantages[5]}"

        assert abs(advantages[4] - 1.1999) < 1e-3, f"A_4期望1.1999，实际{advantages[4]}"
```

```python
            ratio[0] = exp(-1.0 - (-1.0)) = exp(0) = 1.0
            ratio[1] = exp(-0.5 - (-1.0)) = exp(0.5) ≈ 1.6487
            ratio[2] = exp(-1.5 - (-1.0)) = exp(-0.5) ≈ 0.6065
        """
        log_probs = [-1.0, -0.5, -1.5]
        old_log_probs = [-1.0, -1.0, -1.0]
        advantages = [0.5, -0.3, 0.8]
        clip_range = 0.2

        actor_loss, ratios, clipped_ratios = calculate_actor_loss(
            log_probs, old_log_probs, advantages, clip_range,
        )

        # 验证ratio
        assert abs(ratios[0] - 1.0) < 1e-4, f"ratio[0]期望1.0，实际{ratios[0]}"
        assert abs(ratios[1] - math.exp(0.5)) < 1e-4, f"ratio[1]期望{math.exp(0.5):.4f}，实际{ratios[1]}"
        assert abs(ratios[2] - math.exp(-0.5)) < 1e-4, f"ratio[2]期望{math.exp(-0.5):.4f}，实际{ratios[2]}"

    def test_clipping_positive_advantage(self):
        """
        当A > 0时，ratio被裁剪到1+ε
        ratio[1] ≈ 1.6487 > 1.2，应被裁剪为1.2
        """
        log_probs = [-1.0, -0.5]
        old_log_probs = [-1.0, -1.0]
        advantages = [0.5, 0.8]
        clip_range = 0.2

        _, ratios, clipped_ratios = calculate_actor_loss(
            log_probs, old_log_probs, advantages, clip_range,
        )

        # ratio[1] ≈ 1.6487，应被裁剪为1.2
        assert abs(clipped_ratios[1] - 1.2) < 1e-4, f"clipped_ratio[1]期望1.2，实际{clipped_ratios[1]}"

    def test_clipping_negative_advantage(self):
        """
        当A < 0时，ratio被裁剪到1-ε
        ratio = 1.6487, A = -0.3
        clipped_ratio = max(0.8, min(1.2, 1.6487)) = 1.2
        但因为A < 0，min(ratio·A, clipped_ratio·A) = min(1.6487×(-0.3), 1.2×(-0.3))
        = min(-0.4946, -0.36) = -0.4946
        loss = -(-0.4946) = 0.4946
        """
        log_probs = [-0.5]
        old_log_probs = [-1.0]
        advantages = [-0.3]
        clip_range = 0.2

        actor_loss, ratios, clipped_ratios = calculate_actor_loss(
            log_probs, old_log_probs, advantages, clip_range,
        )

        # ratio = exp(0.5) ≈ 1.6487
        ratio = math.exp(0.5)
        assert abs(ratios[0] - ratio) < 1e-4

        # clipped_ratio = 1.2
        assert abs(clipped_ratios[0] - 1.2) < 1e-4

    def test_no_clipping_needed(self):
        """
        ratio在[1-ε, 1+ε]范围内时，不需要裁剪
        """
        log_probs = [-1.0]
        old_log_probs = [-1.0]
        advantages = [0.5]
        clip_range = 0.2

        actor_loss, ratios, clipped_ratios = calculate_actor_loss(
            log_probs, old_log_probs, advantages, clip_range,
        )

        # ratio = 1.0，在[0.8, 1.2]范围内
        assert ratios[0] == clipped_ratios[0]
        # loss = -min(1.0×0.5, 1.0×0.5) = -0.5
        assert abs(actor_loss - (-0.5)) < 1e-6

    def test_length_mismatch_raises(self):
        """输入数组长度不一致应抛出异常"""
        with pytest.raises(ValueError):
            calculate_actor_loss([-1.0], [-1.0, -0.5], [0.5], 0.2)
```

```python
            old_values = [0.5]     (V_t^{old})
            returns    = [0.8]     (R_t)
            clip_range = 0.2

        计算：
            V_clip = clip(1.0, 0.3, 0.7) = 0.7
            loss1 = (1.0 - 0.8)² = 0.04
            loss2 = (0.7 - 0.8)² = 0.01
            max(0.04, 0.01) = 0.04
```
